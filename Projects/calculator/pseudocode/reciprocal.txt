(~ -> Action item)

~reciprocal pressed
    -result var

    -if leftOperand is null
        -leftNumber = solve 1/${entry}
        -display `1/${entry}` to log (don't include anything else)
        -display result to entry
    -else if operator is not null
        -displayToLog(leftOperand, operator, `1/(${entry})`)
        result = solve(1, entry, divide);
        -updateEntryBox(result);
        -rightOperand = result
    -else
        -if clearEntry is false (new number was entered)
            -displayToLog `1/(${entry})`
            -store entry to leftOperand
        -else
            -displayToLog(`1/(${log.textContent})`)
            -leftOperand = (1 / leftOperand)
            -updateEntryBox(leftOperand);
    -set clearEntry to true
